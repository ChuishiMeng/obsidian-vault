# services/template_matcher.py

class TemplateMatcher:
    """模板匹配服务"""
    
    def __init__(self, vector_store, template_repo):
        self.vector_store = vector_store
        self.template_repo = template_repo
    
    async def match(self, floor_type: str, page_type: str, top_k: int = 3) -> List[TemplateMatch]:
        """匹配楼层模板"""
        
        # 1. 向量检索
        query_embedding = await self.vector_store.encode(floor_type)
        candidates = await self.vector_store.search(
            query_embedding, 
            top_k=top_k * 2  # 多取一些候选
        )
        
        # 2. 规则过滤
        filtered = []
        for candidate in candidates:
            template = await self.template_repo.get(candidate.template_id)
            
            # 检查页面类型匹配
            if page_type in template.applicable_page_types:
                # 检查模板状态
                if template.status == 'active':
                    filtered.append(TemplateMatch(
                        template_id=template.id,
                        template_name=template.name,
                        score=candidate.score,
                        preview_url=template.preview_url
                    ))
        
        return filtered[:top_k]
```

class CCPExecutor:
    """CCP 指令执行引擎"""
    
    def __init__(self, legao_api):
        self.legao_api = legao_api
        self.validators = [
            TemplateValidator(),
            SlotValidator(),
            MaterialValidator()
        ]
    
    async def execute(self, instructions: List[CCPInstruction]) -> ExecuteResult:
        """执行指令列表"""
        
        results = []
        for inst in instructions:
            # 1. 校验指令
            validation = await self._validate(inst)
            if not validation.valid:
                results.append(ExecuteResult(
                    instruction_id=inst.instruction_id,
                    success=False,
                    error=validation.error
                ))
                continue
            
            # 2. 翻译并执行
            try:
                api_request = self._translate(inst)
                api_response = await self.legao_api.call(api_request)
                results.append(ExecuteResult(
                    instruction_id=inst.instruction_id,
                    success=True,
                    data=api_response
                ))
            except Exception as e:
                results.append(ExecuteResult(
                    instruction_id=inst.instruction_id,
                    success=False,
                    error=str(e)
                ))
        
        return ExecuteResultBatch(results=results)
    
    def _translate(self, inst: CCPInstruction) -> APIRequest:
        """翻译 CCP 指令为乐高 API"""
        
        if inst.target_type == 'floor' and inst.instruction_type == 'create':
            return APIRequest(
                endpoint='/api/floor/create',
                method='POST',
                body={
                    'page_id': inst.context['page_id'],
                    'template_id': inst.payload['template_id'],
                    'floor_name': inst.payload.get('floor_name', ''),
                    'priority': inst.payload.get('priority', 0)
                }
            )
        # ... 其他指令类型

# services/template_recommender.py

class TemplateRecommender:
    """模板推荐服务"""
    
    async def recommend(
        self,
        query: str,
        page_type: str = None,
        top_k: int = 5
    ) -> List[TemplateRecommendation]:
        """推荐模板"""
        
        # 1. 编码查询
        query_embedding = await self.encoder.encode(query)
        
        # 2. 向量检索
        candidates = await self.vector_store.search(
            query_embedding,
            filter=self._build_filter(page_type),
            top_k=top_k * 2
        )
        
        # 3. 重排序（可选）
        ranked = await self._rerank(candidates, query)
        
        # 4. 返回结果
        return [
            TemplateRecommendation(
                template_id=c.template_id,
                template_name=c.metadata['template_name'],
                score=c.score,
                preview_url=c.metadata.get('preview_url'),
                match_reason=self._explain_match(c, query)
            )
            for c in ranked[:top_k]
        ]
    
    def _explain_match(self, candidate, query) -> str:
        """解释推荐原因"""
        tags = candidate.metadata.get('tags', [])
        matched_tags = [t for t in tags if t in query.lower()]
        if matched_tags:
            return f"匹配标签：{', '.join(matched_tags)}"
        return "语义相似推荐"

# services/image_recommender.py

class ImageRecommender:
    """图片推荐服务"""
    
    def __init__(self, clip_model, vector_store, material_repo):
        self.clip = clip_model
        self.vector_store = vector_store
        self.material_repo = material_repo
    
    async def recommend(
        self,
        slot_type: str,
        page_theme: str,
        top_k: int = 10
    ) -> List[ImageRecommendation]:
        """推荐图片"""
        
        # 1. 构建查询描述
        query_text = f"{page_theme} {slot_type} 图片"
        
        # 2. 文本编码
        text_embedding = await self.clip.encode_text(query_text)
        
        # 3. 向量检索
        candidates = await self.vector_store.search(
            text_embedding,
            collection="images",
            top_k=top_k * 2
        )
        
        # 4. 获取图片详情
        results = []
        for c in candidates:
            material = await self.material_repo.get(c.material_id)
            results.append(ImageRecommendation(
                material_id=material.id,
                image_url=material.url,
                score=c.score,
                tags=material.tags
            ))
        
        return results[:top_k]

# services/page_checker.py

class PageChecker:
    """页面检查服务"""
    
    def __init__(self):
        self.checkers = [
            RequiredFieldChecker(),
            ImageSizeChecker(),
            LinkValidChecker(),
            StrategyConflictChecker(),
            SensitiveWordChecker()
        ]
    
    async def check(self, page_id: str) -> CheckResult:
        """检查页面"""
        
        # 获取页面数据
        page = await self.page_repo.get_full(page_id)
        
        issues = []
        for checker in self.checkers:
            checker_issues = await checker.check(page)
            issues.extend(checker_issues)
        
        return CheckResult(
            page_id=page_id,
            total_issues=len(issues),
            errors=[i for i in issues if i.level == 'error'],
            warnings=[i for i in issues if i.level == 'warning'],
            passed=len([i for i in issues if i.level == 'error']) == 0
        )

class RequiredFieldChecker:
    """必填项检查器"""
    
    async def check(self, page: Page) -> List[Issue]:
        issues = []
        
        for floor in page.floors:
            template = await self.template_repo.get(floor.template_id)
            
            for slot_def in template.slots:
                if slot_def.required:
                    slot = floor.get_slot(slot_def.slot_id)
                    if not slot or not slot.materials:
                        issues.append(Issue(
                            level='error',
                            type='required_field_missing',
                            message=f"楼层[{floor.name}]的坑位[{slot_def.name}]未配置素材",
                            location=f"floor.{floor.id}.slot.{slot_def.slot_id}"
                        ))
        
        return issues

        historical_pages = await self.get_similar_pages(page_type)
        
        # 2. 分析人群分布
        crowd_dist = self.analyze_crowd_distribution(historical_pages)
        
        # 3. 分析时间分布
        time_dist = self.analyze_time_distribution(historical_pages)
        
        # 4. 生成推荐
        return StrategyRecommendation(
            recommended_crowds=self._get_top_crowds(crowd_dist, top_k=3),
            recommended_time_slots=self._get_top_times(time_dist, top_k=3),
            recommended_priority=self._estimate_priority(page_type),
            confidence=0.75
        )
    
    def analyze_crowd_distribution(self, pages: List[Page]) -> Dict:
        """分析人群分布"""
        crowd_counts = defaultdict(int)
        for page in pages:
            for strategy in page.strategies:
                for crowd in strategy.crowds:
                    crowd_counts[crowd.id] += 1
        
        total = sum(crowd_counts.values())
        return {
            crowd_id: count / total 
            for crowd_id, count in crowd_counts.items()
        }
```

### 4.1 意图理解准确率

**问题**：用户描述模糊，AI 理解不准确

**解决方案**：

| 方案 | 说明 |
|------|------|
| **多轮澄清** | 意图不明确时，主动提问澄清 |
| **预设模板** | 提供常见场景模板，降低输入门槛 |
| **Few-shot学习** | 使用标注数据优化 Prompt |
| **反馈学习** | 收集用户修正数据，持续优化 |

### 4.3 生成内容质量

**问题**：AI 生成的文案质量参差不齐

**解决方案**：

| 方案 | 说明 |
|------|------|
| **风格控制** | Prompt 中加入品牌调性要求 |

| **缓存优化** | 缓存常见意图的结构模板 |
